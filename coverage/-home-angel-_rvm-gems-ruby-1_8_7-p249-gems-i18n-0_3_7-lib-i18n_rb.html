134     # Translates, pluralizes and interpolates a given key using a given locale,
 
135     # scope, and default, as well as interpolation values.
 
136     #
 
137     # *LOOKUP*
 
138     #
 
139     # Translation data is organized as a nested hash using the upper-level keys
 
140     # as namespaces. <em>E.g.</em>, ActionView ships with the translation:
 
141     # <tt>:date => {:formats => {:short => "%b %d"}}</tt>.
 
142     #
 
143     # Translations can be looked up at any level of this hash using the key argument
 
144     # and the scope option. <em>E.g.</em>, in this example <tt>I18n.t :date</tt>
 
145     # returns the whole translations hash <tt>{:formats => {:short => "%b %d"}}</tt>.
 
146     #
 
147     # Key can be either a single key or a dot-separated key (both Strings and Symbols
 
148     # work). <em>E.g.</em>, the short format can be looked up using both:
 
149     #   I18n.t 'date.formats.short'
 
150     #   I18n.t :'date.formats.short'
 
151     #
 
152     # Scope can be either a single key, a dot-separated key or an array of keys
 
153     # or dot-separated keys. Keys and scopes can be combined freely. So these
 
154     # examples will all look up the same short date format:
 
155     #   I18n.t 'date.formats.short'
 
156     #   I18n.t 'formats.short', :scope => 'date'
 
157     #   I18n.t 'short', :scope => 'date.formats'
 
158     #   I18n.t 'short', :scope => %w(date formats)
 
159     #
 
160     # *INTERPOLATION*
 
161     #
 
162     # Translations can contain interpolation variables which will be replaced by
 
163     # values passed to #translate as part of the options hash, with the keys matching
 
164     # the interpolation variable names.
 
165     #
 
166     # <em>E.g.</em>, with a translation <tt>:foo => "foo {{bar}}"</tt> the option
 
167     # value for the key +bar+ will be interpolated into the translation:
 
168     #   I18n.t :foo, :bar => 'baz' # => 'foo baz'
 
169     #
 
170     # *PLURALIZATION*
 
171     #
 
172     # Translation data can contain pluralized translations. Pluralized translations
 
173     # are arrays of singluar/plural versions of translations like <tt>['Foo', 'Foos']</tt>.
 
174     #
 
175     # Note that <tt>I18n::Backend::Simple</tt> only supports an algorithm for English
 
176     # pluralization rules. Other algorithms can be supported by custom backends.
 
177     #
 
178     # This returns the singular version of a pluralized translation:
 
179     #   I18n.t :foo, :count => 1 # => 'Foo'
 
180     #
 
181     # These both return the plural version of a pluralized translation:
 
182     #   I18n.t :foo, :count => 0 # => 'Foos'
 
183     #   I18n.t :foo, :count => 2 # => 'Foos'
 
184     #
 
185     # The <tt>:count</tt> option can be used both for pluralization and interpolation.
 
186     # <em>E.g.</em>, with the translation
 
187     # <tt>:foo => ['{{count}} foo', '{{count}} foos']</tt>, count will
 
188     # be interpolated to the pluralized translation:
 
189     #   I18n.t :foo, :count => 1 # => '1 foo'
 
190     #
 
191     # *DEFAULTS*
 
192     #
 
193     # This returns the translation for <tt>:foo</tt> or <tt>default</tt> if no translation was found:
 
194     #   I18n.t :foo, :default => 'default'
 
195     #
 
196     # This returns the translation for <tt>:foo</tt> or the translation for <tt>:bar</tt> if no
 
197     # translation for <tt>:foo</tt> was found:
 
198     #   I18n.t :foo, :default => :bar
 
199     #
 
200     # Returns the translation for <tt>:foo</tt> or the translation for <tt>:bar</tt>
 
201     # or <tt>default</tt> if no translations for <tt>:foo</tt> and <tt>:bar</tt> were found.
 
202     #   I18n.t :foo, :default => [:bar, 'default']
 
203     #
 
204     # *BULK LOOKUP*
 
205     #
 
206     # This returns an array with the translations for <tt>:foo</tt> and <tt>:bar</tt>.
 
207     #   I18n.t [:foo, :bar]
 
208     #
 
209     # Can be used with dot-separated nested keys:
 
210     #   I18n.t [:'baz.foo', :'baz.bar']
 
211     #
 
212     # Which is the same as using a scope option:
 
213     #   I18n.t [:foo, :bar], :scope => :baz
 
214     #
 
215     # *LAMBDAS*
 
216     #
 
217     # Both translations and defaults can be given as Ruby lambdas. Lambdas will be
 
218     # called and passed the key and options.
 
219     #
 
220     # E.g. assuming the key <tt>:salutation</tt> resolves to:
 
221     #   lambda { |key, options| options[:gender] == 'm' ? "Mr. {{options[:name]}}" : "Mrs. {{options[:name]}}"Â }
 
222     #
 
223     # Then <tt>I18n.t(:salutation, :gender => 'w', :name => 'Smith') will result in "Mrs. Smith".
 
224     #
 
225     # It is recommended to use/implement lambdas in an "idempotent" way. E.g. when
 
226     # a cache layer is put in front of I18n.translate it will generate a cache key
 
227     # from the argument values passed to #translate. Therefor your lambdas should
 
228     # always return the same translations/values per unique combination of argument
 
229     # values.
 
230     def translate(*args)
 
231       options = args.pop if args.last.is_a?(Hash)
 
232       key     = args.shift
 
233       locale  = options && options.delete(:locale) || config.locale
 
234       raises  = options && options.delete(:raise)
 
235       config.backend.translate(locale, key, options || {})
 
236     rescue I18n::ArgumentError => exception
 
237       raise exception if raises
 
238       handle_exception(exception, locale, key, options)
 
239     end

278     # Any exceptions thrown in translate will be sent to the @@exception_handler
 
279     # which can be a Symbol, a Proc or any other Object.
 
280     #
 
281     # If exception_handler is a Symbol then it will simply be sent to I18n as
 
282     # a method call. A Proc will simply be called. In any other case the
 
283     # method #call will be called on the exception_handler object.
 
284     #
 
285     # Examples:
 
286     #
 
287     #   I18n.exception_handler = :default_exception_handler             # this is the default
 
288     #   I18n.default_exception_handler(exception, locale, key, options) # will be called like this
 
289     #
 
290     #   I18n.exception_handler = lambda { |*args| ... }                 # a lambda
 
291     #   I18n.exception_handler.call(exception, locale, key, options)    # will be called like this
 
292     #
 
293     #  I18n.exception_handler = I18nExceptionHandler.new                # an object
 
294     #  I18n.exception_handler.call(exception, locale, key, options)     # will be called like this
 
295     def handle_exception(exception, locale, key, options)
 
296       case config.exception_handler
 
297       when Symbol
 
298         send(config.exception_handler, exception, locale, key, options)
 
299       else
 
300         config.exception_handler.call(exception, locale, key, options)
 
301       end
 
302     end

311       normalized_key_cache(separator)[key] ||=
 
312         case key
 
313         when Array
