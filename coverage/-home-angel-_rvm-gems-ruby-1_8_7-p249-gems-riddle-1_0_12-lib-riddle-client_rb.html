318     # Build excerpts from search terms (the +words+) and the text of documents. Excerpts are bodies of text that have the +words+ highlighted.
 
319     # They may also be abbreviated to fit within a word limit.
 
320     #
 
321     # As part of the options hash, you will need to
 
322     # define:
 
323     # * :docs
 
324     # * :words
 
325     # * :index
 
326     #
 
327     # Optional settings include:
 
328     # * :before_match (defaults to <span class="match">)
 
329     # * :after_match (defaults to </span>)
 
330     # * :chunk_separator (defaults to ' &#8230; ' - which is an HTML ellipsis)
 
331     # * :limit (defaults to 256)
 
332     # * :around (defaults to 5)
 
333     # * :exact_phrase (defaults to false)
 
334     # * :single_passage (defaults to false)
 
335     #
 
336     # The defaults differ from the official PHP client, as I've opted for
 
337     # semantic HTML markup.
 
338     #
 
339     # Example:
 
340     #
 
341     #   client.excerpts(:docs => ["Pat Allan, Pat Cash"], :words => 'Pat', :index => 'pats')
 
342     #   #=> ["<span class=\"match\">Pat</span> Allan, <span class=\"match\">Pat</span> Cash"]
 
343     #
 
344     #   lorem_lipsum = "Lorem ipsum dolor..."

346     #   client.excerpts(:docs => ["Pat Allan, #{lorem_lipsum} Pat Cash"], :words => 'Pat', :index => 'pats')
 
347     #   #=> ["<span class=\"match\">Pat</span> Allan, Lorem ipsum dolor sit amet, consectetur adipisicing
 
348     #          elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua &#8230; . Excepteur 
349     #          sint occaecat cupidatat non proident, sunt in culpa qui officia deserunt mollit anim id est 
350     #          laborum. <span class=\"match\">Pat</span> Cash"]

544     # Send a collection of messages, for a command type (eg, search, excerpts,
 
545     # update), to the Sphinx daemon.
 
546     def request(command, messages)
 
547       response = ""
 
548       status   = -1
 
549       version  = 0
 
550       length   = 0
 
551       message  = Array(messages).join("")
 
552       if message.respond_to?(:force_encoding)
 
553         message = message.force_encoding('ASCII-8BIT')
 
554       end
 
555       
556       connect do |socket|
 
557         case command
 
558         when :search
 
559           # Message length is +4 to account for the following count value for
 
560           # the number of messages (well, that's what I'm assuming).
 
561           socket.send [
 
562             Commands[command], Versions[command],
 
563             4+message.length,  messages.length
 
564           ].pack("nnNN") + message, 0
 
565         when :status
 
566           socket.send [
 
567             Commands[command], Versions[command], 4, 1
 
568           ].pack("nnNN"), 0
 
569         else
 
570           socket.send [
 
571             Commands[command], Versions[command], message.length
 
572           ].pack("nnN") + message, 0
 
573         end
 
574         
575         header = socket.recv(8)
 
576         status, version, length = header.unpack('n2N')
 
577         
578         while response.length < (length || 0)
 
579           part = socket.recv(length - response.length)
 
580           response << part if part
 
581         end
 
582       end
 
583       
584       if response.empty? || response.length != length
 
585         raise ResponseError, "No response from searchd (status: #{status}, version: #{version})"
 
586       end
 
587       
588       case status
 
589       when Statuses[:ok]
 
590         if version < Versions[command]
 
591           puts format("searchd command v.%d.%d older than client (v.%d.%d)",
 
592             version >> 8, version & 0xff,
 
593             Versions[command] >> 8, Versions[command] & 0xff)
 
594         end
 
595         response
 
596       when Statuses[:warning]
 
597         length = response[0, 4].unpack('N*').first
 
598         puts response[4, length]
 
599         response[4 + length, response.length - 4 - length]
 
600       when Statuses[:error], Statuses[:retry]
 
601         raise ResponseError, "searchd error (status: #{status}): #{response[4, response.length - 4]}"
 
602       else
 
603         raise ResponseError, "Unknown searchd error (status: #{status})"
 
604       end
 
605     end
