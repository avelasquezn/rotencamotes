189       # Defines an attributes writer for the specified association(s). If you

193       # Supported options:
 
194       # [:allow_destroy]
 
195       #   If true, destroys any members from the attributes hash with a
 
196       #   <tt>_destroy</tt> key and a value that evaluates to +true+

198       # [:reject_if]

201       #   hash. The hash is passed to the supplied Proc or the method
 
202       #   and it should return either +true+ or +false+. When no :reject_if

271     # If update_only is false and the given attributes include an <tt>:id</tt>

273     # modified. If update_only is true, a new record is only created when no
 
274     # object exists. Otherwise a new record will be built.

301     # Assigns the given attributes to the collection association.
 
302     #
 
303     # Hashes with an <tt>:id</tt> value matching an existing associated record
 
304     # will update that record. Hashes without an <tt>:id</tt> value will build
 
305     # a new record for the association. Hashes with a matching <tt>:id</tt>
 
306     # value and a <tt>:_destroy</tt> key set to a truthy value will mark the
 
307     # matched record for destruction.
 
308     #
 
309     # For example:
 
310     #
 
311     #   assign_nested_attributes_for_collection_association(:people, {
 
312     #     '1' => { :id => '1', :name => 'Peter' },
 
313     #     '2' => { :name => 'John' },
 
314     #     '3' => { :id => '2', :_destroy => true }

363           raise_nested_attributes_record_not_found(association_name, attributes['id'])
