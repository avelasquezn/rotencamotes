123       # Conceptually, this expands "parenthesized selectors".
 
124       # That is, if we have `.A .B {@extend .C}` and `.D .C {...}`,
 
125       # this conceptually expands into `.D .C, .D (.A .B)`,
 
126       # and this function translates `.D (.A .B)` into `.D .A .B, .A.D .B, .D .A .B`.
 
127       #
 
128       # @param path [Array<Array<SimpleSequence or String>>] A list of parenthesized selector groups.
 
129       # @return [Array<Array<SimpleSequence or String>>] A list of fully-expanded selectors.
 
130       def weave(path)
 
131         befores = [[]]
 
132         afters = path.dup

135           current = afters.shift.dup
 
136           last_current = [current.pop]

138             last_current.unshift(current.pop)

141               subweave(before, current).map {|seqs| seqs + last_current}

150       #
 
151       # For example, given `.foo .bar` and `.baz .bang`,

209       def subweave_superselector?(sseq1, sseq2)
 
210         if sseq1.size > 1
 
211           # More complex selectors are never superselectors of less complex ones
 
212           return unless sseq2.size > 1
 
213           # .foo ~ .bar is a superselector of .foo + .bar
 
214           return unless sseq1[1] == "~" ? sseq2[1] != ">" : sseq2[1] == sseq1[1]
