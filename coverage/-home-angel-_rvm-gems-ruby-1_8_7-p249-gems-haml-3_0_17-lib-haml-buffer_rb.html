234     # Merges two attribute hashes.
 
235     # This is the same as `to.merge!(from)`,
 
236     # except that it merges id, class, and data attributes.
 
237     #
 
238     # ids are concatenated with `"_"`,
 
239     # and classes are concatenated with `" "`.
 
240     # data hashes are simply merged.
 
241     #
 
242     # Destructively modifies both `to` and `from`.
 
243     #
 
244     # @param to [{String => String}] The attribute hash to merge into
 
245     # @param from [{String => #to_s}] The attribute hash to merge from
 
246     # @return [{String => String}] `to`, after being merged
 
247     def self.merge_attrs(to, from)
 
248       from['id'] = Precompiler.filter_and_join(from['id'], '_') if from['id']
 
249       if to['id'] && from['id']
 
250         to['id'] << '_' << from.delete('id').to_s
 
251       elsif to['id'] || from['id']
 
252         from['id'] ||= to['id']
 
253       end
 
254 
255       from['class'] = Precompiler.filter_and_join(from['class'], ' ') if from['class']
 
256       if to['class'] && from['class']
 
257         # Make sure we don't duplicate class names
 
258         from['class'] = (from['class'].to_s.split(' ') | to['class'].split(' ')).sort.join(' ')
 
259       elsif to['class'] || from['class']
 
260         from['class'] ||= to['class']
 
261       end
 
262 
263       from_data = from['data'].is_a?(Hash)
 
264       to_data = to['data'].is_a?(Hash)
 
265       if from_data && to_data
 
266         to['data'] = to['data'].merge(from['data'])
 
267       elsif to_data
 
268         to = Haml::Util.map_keys(to.delete('data')) {|name| "data-#{name}"}.merge(to)
 
269       elsif from_data
 
270         from = Haml::Util.map_keys(from.delete('data')) {|name| "data-#{name}"}.merge(from)
 
271       end
 
272 
273       to.merge!(from)
 
274     end

289     def parse_object_ref(ref)
