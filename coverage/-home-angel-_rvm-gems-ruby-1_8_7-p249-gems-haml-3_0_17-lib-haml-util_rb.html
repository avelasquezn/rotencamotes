640     # This is used for methods in {Haml::Buffer} that need to be very fast,
 
641     # and take a lot of boolean parameters
 
642     # that are known at compile-time.
 
643     # Instead of passing the parameters in normally,
 
644     # a separate method is defined for every possible combination of those parameters;
 
645     # these are then called using \{#static\_method\_name}.
 
646     #
 
647     # To define a static method, an ERB template for the method is provided.
 
648     # All conditionals based on the static parameters
 
649     # are done as embedded Ruby within this template.
 
650     # For example:
 
651     #
 
652     #     def_static_method(Foo, :my_static_method, [:foo, :bar], :baz, :bang, <<RUBY)
 
653     #       <% if baz && bang %>
 
654     #         return foo + bar
 
655     #       <% elsif baz || bang %>
 
656     #         return foo - bar
 
657     #       <% else %>
 
658     #         return 17
 
659     #       <% end %>
 
660     #     RUBY
 
661     #
 
662     # \{#static\_method\_name} can be used to call static methods.
 
663     #
 
664     # @overload def_static_method(klass, name, args, *vars, erb)
 
665     # @param klass [Module] The class on which to define the static method
 
666     # @param name [#to_s] The (base) name of the static method
 
667     # @param args [Array<Symbol>] The names of the arguments to the defined methods
 
668     #   (**not** to the ERB template)
 
669     # @param vars [Array<Symbol>] The names of the static boolean variables
 
670     #   to be made available to the ERB template
 
671     # @param erb [String] The template for the method code
 
672     def def_static_method(klass, name, args, *vars)
 
673       erb = vars.pop
 
674       info = caller_info
 
675       powerset(vars).each do |set|
 
676         context = StaticConditionalContext.new(set).instance_eval {binding}
 
677         klass.class_eval(<<METHOD, info[0], info[1])
 
678 def #{static_method_name(name, *vars.map {|v| set.include?(v)})}(#{args.join(', ')})
 
679   #{ERB.new(erb).result(context)}
 
680 end
 
681 METHOD
 
682       end
 
683     end

696     # Calculates the memoization table for the Least Common Subsequence algorithm.
 
697     # Algorithm from [Wikipedia](http://en.wikipedia.org/wiki/Longest_common_subsequence_problem#Computing_the_length_of_the_LCS)
 
698     def lcs_table(x, y)
 
699       c = Array.new(x.size) {[]}
 
700       x.size.times {|i| c[i][0] = 0}
 
701       y.size.times {|j| c[0][j] = 0}
 
702       (1...x.size).each do |i|
 
703         (1...y.size).each do |j|
 
704           c[i][j] =
 
705             if yield x[i], y[j]
 
706               c[i-1][j-1] + 1
 
707             else
 
708               [c[i][j-1], c[i-1][j]].max
 
709             end
 
710         end
 
711       end
 
712       return c
 
713     end

715     # Computes a single longest common subsequence for arrays x and y.
 
716     # Algorithm from [Wikipedia](http://en.wikipedia.org/wiki/Longest_common_subsequence_problem#Reading_out_an_LCS)
 
717     def lcs_backtrace(c, x, y, i, j, &block)
 
718       return [] if i == 0 || j == 0
 
719       if v = yield(x[i], y[j])
 
720         return lcs_backtrace(c, x, y, i-1, j-1, &block) << v
 
721       end
 
722 
723       return lcs_backtrace(c, x, y, i, j-1, &block) if c[i][j-1] > c[i-1][j]
 
724       return lcs_backtrace(c, x, y, i-1, j, &block)
 
725     end

727     # Parses a magic comment at the beginning of a Haml file.
 
728     # The parsing rules are basically the same as Ruby's.
 
729     #
 
730     # @return [(Boolean, String or nil)]
 
731     #   Whether the document begins with a UTF-8 BOM,
 
732     #   and the declared encoding of the document (or nil if none is declared)
 
733     def parse_haml_magic_comment(str)
 
734       scanner = StringScanner.new(str.dup.force_encoding("BINARY"))
 
735       bom = scanner.scan(/\xEF\xBB\xBF/n)
 
736       return bom unless scanner.scan(/-\s*#\s*/n)
 
737       if coding = try_parse_haml_emacs_magic_comment(scanner)
 
738         return bom, coding
 
739       end
 
740 
741       return bom unless scanner.scan(/.*?coding[=:]\s*([\w-]+)/in)
 
742       return bom, scanner[1]
 
743     end

745     def try_parse_haml_emacs_magic_comment(scanner)
 
746       pos = scanner.pos
 
747       return unless scanner.scan(/.*?-\*-\s*/n)
 
748       # From Ruby's parse.y
 
749       return unless scanner.scan(/([^\s'":;]+)\s*:\s*("(?:\\.|[^"])*"|[^"\s;]+?)[\s;]*-\*-/n)
 
750       name, val = scanner[1], scanner[2]
 
751       return unless name =~ /(en)?coding/in
 
752       val = $1 if val =~ /^"(.*)"$/n
 
753       return val
 
754     ensure
 
755       scanner.pos = pos
 
756     end
