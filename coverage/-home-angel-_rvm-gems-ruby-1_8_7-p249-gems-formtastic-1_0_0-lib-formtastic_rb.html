1005       def date_or_datetime_input(method, options)
 
1006         position = { :year => 1, :month => 2, :day => 3, :hour => 4, :minute => 5, :second => 6 }
 
1007         i18n_date_order = ::I18n.t(:order, :scope => [:date])
 
1008         i18n_date_order = nil unless i18n_date_order.is_a?(Array)
 
1009         inputs   = options.delete(:order) || i18n_date_order || [:year, :month, :day]
 
1010         inputs   = [] if options[:ignore_date]
 
1011         labels   = options.delete(:labels) || {}
 
1012 
1013         time_inputs = [:hour, :minute]
 
1014         time_inputs << :second if options[:include_seconds]
 
1015 
1016         list_items_capture = ""
 
1017         hidden_fields_capture = ""
 
1018 
1019         datetime = options[:selected]
 
1020         datetime = @object.send(method) if @object && @object.send(method) # object value trumps :selected value
 
1021 
1022         html_options = options.delete(:input_html) || {}
 
1023         input_ids    = []
 
1024 
1025         (inputs + time_inputs).each do |input|
 
1026           input_ids << input_id = generate_html_id(method, "#{position[input]}i")
 
1027 
1028           field_name = "#{method}(#{position[input]}i)"
 
1029           if options[:"discard_#{input}"]
 
1030             break if time_inputs.include?(input)
 
1031 
1032             hidden_value = datetime.respond_to?(input) ? datetime.send(input) : datetime
 
1033             hidden_fields_capture << template.hidden_field_tag("#{@object_name}[#{field_name}]", (hidden_value || 1), :id => input_id)

1459         method_to_group_association = method_class.reflect_on_association(group_by)
