715     def instances_from_class(klass, matches)
 
716       index_options = klass.sphinx_index_options
 
717 
718       ids = matches.collect { |match| match[:attributes]["sphinx_internal_id"] }
 
719       instances = ids.length > 0 ? klass.find(
 
720         :all,
 
721         :joins      => options[:joins],
 
722         :conditions => {klass.primary_key_for_sphinx.to_sym => ids},
 
723         :include    => (options[:include] || index_options[:include]),
 
724         :select     => (options[:select]  || index_options[:select]),
 
725         :order      => (options[:sql_order] || index_options[:sql_order])
 
726       ) : []
 
727 
728       # Raise an exception if we find records in Sphinx but not in the DB, so
 
729       # the search method can retry without them. See 
730       # ThinkingSphinx::Search.retry_search_on_stale_index.
 
731       if options[:raise_on_stale] && instances.length < ids.length
 
732         stale_ids = ids - instances.map { |i| i.id }
 
733         raise StaleIdsException, stale_ids
 
734       end
 
735 
736       # if the user has specified an SQL order, return the collection
 
737       # without rearranging it into the Sphinx order
 
738       return instances if (options[:sql_order] || index_options[:sql_order])
 
739 
740       ids.collect { |obj_id|
 
741         instances.detect do |obj|
 
742           obj.primary_key_for_sphinx == obj_id
 
743         end
 
744       }

751       return single_class_results if one_class

753       groups = results[:matches].group_by { |match|
 
754         match[:attributes]["class_crc"]

801       search.args.each { |arg| args << arg }
 
802       
803       search.options.keys.each do |key|
